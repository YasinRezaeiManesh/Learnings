groups = {
    1xx ==> Formational (تشکیلی),
    2xx ==> Successful (تایید شده),
    3xx ==> Redirection (تغییر مسیر),
    4xx ==> Error (خطا),
    5xx ==> Server Error (خطای سرور)
}

1xx = {
    100 ==> Continue : سرور بخش اولیه درخواست را دریافت کرده و اماده است تا کلاینت بقیه بدنه را ارسال کند ,
    101 ==> Switching Protocols : سرور موافقت میکند که به پروتوکل دیگری تغییر کند ,
    102 ==> Processing : سرور در حال پردازش درخواست است اما هنوز پاسخی در دسترس نیست ,
    103 ==> Early Hints : نکاتی را در مورد منابعی که ممکن است کلاینت برای شروع پیش‌بارگذاری نیاز داشته باشد، در حالی که سرور هنوز در حال پردازش درخواست اصلی است، ارائه می‌دهد. این می‌تواند به بهبود زمان بارگذاری صفحه کمک کند ,
    104 ==> Connection Reset By Pear : ارتباط بین کلاینت و سرور توسط سرور قطع شده است
}

2xx = {
    200 ==> OK : پاسخ استاندارد موفقیت که نشان میدهد درخواست موفقیت آمیز بوده و سرور منبع درخواستی را بازگردانده است ,
    201 ==> Created : درخواست با موفیقت انجام شد و یک منبع جدید روی سرور ایجاد شد ,
    202 ==> Accepted : درخواست برای پردازش پذیرفته شده است اما پردازش هنوز تکمیل نشده ، این کد اغلب برای عملیات های ناهمزمان استفاده میشود ,
    203 ==> Non Authoritative Information : نشان میدهد که درخواست موفقیت آمیز بوده اما هدر یا بدنه پاسخ قبل از ارسال به کلاینت توسط پروکسی تبدیل کننده تغییر داده شده اند ,
    204 ==> Non Content : درخواست موفقیت آمیز بود اما هیچ محتوایی برای ارسال در پاسخ وجود ندارد
}
